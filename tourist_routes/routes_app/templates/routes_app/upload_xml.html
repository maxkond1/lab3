Поддерживаемые структуры XML:

Стандартная структура:

<tourist_routes>
    <route>
        <name>Название</name>
        <description>Описание</description>
        <length_km>42.5</length_km>
        <duration_days>7</duration_days>
        <difficulty>сложный</difficulty>
        <region>Регион</region>
        <best_season>июль-август</best_season>
    </route>
</tourist_routes>

Альтернативная структура:

<routes>
    <tourist_route>
        <title>Название</title>
        <desc>Описание</desc>
        <distance>42.5</distance>
        <days>7</days>
        <complexity>сложный</complexity>
        <location>Регион</location>
    </tourist_route>
</routes>
